First table: example read with simple sonification

This is iterating over the table in top to bottom for each column:

0.9 0.4 1.4
0.9 0.5 1.4
0.9 0.5 1.3
1.0 0.6 1.4
1.0 0.6 1.3

[0.9, 0.4, 1.4], [0.9, 0.5, 1.4], …

Each column tunes a oscillator at frequency \(200(x + 100)\) and the three are played in parallel.

Each column is normalized to 0-1 based on a per column local calculation of \(z_i = x_i-min(x) / max(x)-min(x)\)

  • Parameters
    • data: the data read from file
    • crnt: current value
    • prevA-C: previous value
    • waitTime: the time it takes between two values (1)
    • normalize: normalization function
    • colMin: minimum value of the table

    • colMax: maximum value of the table 

      (
var r, data, crnt, prevA=0, prevB=0, prevC=0, waitTime=1, normalize, colMax=1.6, colMin=0.4;
data = CSVFileReader.read("/path/to/table1.csv");
data.size.do({arg ndx;
    data[ndx].removeAt(0);
});

normalize = { arg x, min, max; (x - min) / (max - min); };

r = Routine {
    data.size.do({ arg n;
    Routine {
        data[n].yield;
    }.idle(1);
});
};

fork {
    loop {
    crnt = r.value.asFloat;
    "Current vector: ".post;
    crnt.postln;
    "-----------".postln;
    a = normalize.value(crnt[0], colMin, colMax);
    b = normalize.value(crnt[1], colMin, colMax);
    c = normalize.value(crnt[2], colMin, colMax);
    /////////////////////////////
    // Silly example synthesis //
    { [ SinOsc.ar(Line.kr(prevA, a, waitTime, 200, 100, 2), 0, 0.2),
        SinOsc.ar(Line.ar(prevB, b, waitTime, 200, 100, 2), 0, 0.2),
        SinOsc.ar(Line.ar(prevC, c, waitTime, 200, 100, 2), 0, 0.2)] }.play;
       /////////////////////////////
       waitTime.wait;
       prevA = a;
       prevB = b;
       prevC = c;
   }
}
)
